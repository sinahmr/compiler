$Declaration → $FunDeclaration

$FunDeclaration → void #set_ret_type ID #def_func #start_scope ( $Params ) #init_func $CompoundStmt #end_scope

$ParamList → $Param #func_add_param

$Param → int ID #def_arr [ ]
$CompoundStmt → { #start_scope $LocalDeclarations $StatementList #end_scope }

$LocalDeclarations → ε
$StatementList → $StatementList $Statement
$StatementList → ε
$Statement → $ExpressionStmt
$Statement → $CompoundStmt

$Statement → output ( $Args ) ;
$ExpressionStmt → $Var #pid = $Expression #assign ;
$ExpressionStmt → ;

$SelectionStmt → if ( $GenExpression ) $Statement else $Statement
$IterationStmt → while ( $GenExpression ) $Statement
$ReturnStmt → return ;
$ReturnStmt → return $GenExpression #set_ret_value ;
$Var → ID #pid
$Var → ID #pid #push_arr_size [ $Expression ] #arr_value
$GenExpression → $RelExpression
$GenExpression → $Expression
$RelExpression → $RelExpression && $RelTerm
$RelExpression → $RelTerm

$RelTerm → $Expression < $Expression #larger
$Expression → $Expression $AddOp $Term
$Expression → $Term
$AddOp → +
$AddOp → -
$Term → $Term $MulOp $Factor
$Term → $Factor
$MulOp → *
$MulOp → /
